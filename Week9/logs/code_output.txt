import sqlite3, csv, io, re

csv_data = "product,price,quantity\nLaptop,$1000,1\nSmartphone,$800,2\nT-Shirt,$15,3\nHeadphones,$60,4\nTablet,$700,5"

f = io.StringIO(csv_data.strip())
reader = csv.reader(f)
header = [h.strip() for h in next(reader)]
first_row = next(reader, None)

conn = sqlite3.connect("/home/shreyasinghal/Desktop/Training Bootcamp/Week9/data/business.db")
cur  = conn.cursor()
cur.execute("DROP TABLE IF EXISTS data_import")

def guess_type(v):
    if not v: return "TEXT"
    try: float(str(v).replace(",","").strip()); return "REAL"
    except: return "TEXT"

col_defs = [f'"{h}" {guess_type(first_row[i] if first_row and i < len(first_row) else "")}' for i, h in enumerate(header)]
cur.execute(f'CREATE TABLE data_import ({", ".join(col_defs)})')

f.seek(0); next(reader)
ok = err = 0
for row in reader:
    row = [re.sub(r'["\\+]', '', c).strip() for c in row]
    row = [" ".join(c.split()) for c in row]
    if len(row) > len(header): row = row[:len(header)]
    elif len(row) < len(header): row += [""] * (len(header) - len(row))
    try:
        cur.execute(f'INSERT INTO data_import VALUES ({",".join(["?"]*len(header))})', row)
        ok += 1
    except: err += 1

conn.commit(); conn.close()
