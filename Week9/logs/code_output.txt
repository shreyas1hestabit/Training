data = "product,Fabric,Quantity,price\nShirt,Cotton,100,15.99\nPants,Denim,50,29.99\nDress,Silk,20,49.99\nJacket,Leather,30,69.99\nScarf,Wool,40,9.99\nHat,Cotton,60,12.99\nGloves,Leather,10,19.99\nShoes,Rubber,80,39.99\nSocks,Cotton,120,6.99\nBelt,Leather,25,14.99"

reader = csv.reader(io.StringIO(data))

next(reader)  # Skip header

expensive_products = []

for row in reader:
    try:
        price = float(row[3])
        if price > 49.00:
            expensive_products.append(row[0])
    except Exception as e:
        print(f"Error: {e}")

print("Expensive products:")
for i, product in enumerate(expensive_products, start=1):
    print(f"{i}. {product}")
